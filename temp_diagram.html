usecaseDiagram
    actor "Khách (Guest)" as Guest
    actor "Quản lý (Manager)" as Manager
    actor "Quản trị viên (Admin)" as Admin

    package "Frontend (Website)" {
        usecase "Xem Trang chủ & Giới thiệu" as UC_ViewHome
        usecase "Tra cứu Thông tin (Du học/XKLĐ)" as UC_ViewPrograms
        usecase "Đọc Tin tức & Sự kiện" as UC_ViewNews
        usecase "Tìm kiếm nội dung" as UC_Search
        usecase "Gửi Đăng ký Tư vấn" as UC_Register
        usecase "Gửi Liên hệ" as UC_Contact
    }

    package "Backend (Admin Panel)" {
        usecase "Đăng nhập Dashboard" as UC_Login
        
        %% Nhóm Quản lý Nội dung (CMS & News)
        usecase "Quản lý Tin tức (Thêm/Sửa/Xóa)" as UC_ManageNews
        usecase "Quản lý Nội dung CMS (Text/Banner)" as UC_ManageCMS
        usecase "Quản lý Content Blocks" as UC_ManageBlocks
        usecase "Ẩn/Hiện Section trang chủ" as UC_ToggleSection

        %% Nhóm Quản lý Khách hàng
        usecase "Xem Danh sách Đăng ký" as UC_ViewRegistrations
        usecase "Xuất báo cáo (Excel)" as UC_ExportData

        %% Nhóm Quản trị Hệ thống (Admin Only)
        usecase "Quản lý Tài khoản (Users)" as UC_ManageUsers
        usecase "Phân quyền (Permissions)" as UC_ManagePermissions
        usecase "Cài đặt Hệ thống (Settings)" as UC_SystemSettings
        usecase "Xem Logs hệ thống" as UC_ViewLogs
    }

    %% Guest Relationships
    Guest --> UC_ViewHome
    Guest --> UC_ViewPrograms
    Guest --> UC_ViewNews
    Guest --> UC_Search
    Guest --> UC_Register
    Guest --> UC_Contact

    %% Manager Relationships (Core Business)
    Manager --> UC_Login
    Manager --> UC_ManageNews
    Manager --> UC_ManageCMS
    Manager --> UC_ManageBlocks
    Manager --> UC_ViewRegistrations
    Manager --> UC_ExportData

    %% Admin Relationships (System Owner)
    Admin --|> Manager : Kế thừa quyền
    Admin --> UC_ManageUsers
    Admin --> UC_ManagePermissions
    Admin --> UC_SystemSettings
    Admin --> UC_ViewLogs

    %% Includes & Extends
    UC_ManageCMS ..> UC_ToggleSection : include
    UC_ManageUsers ..> UC_ManagePermissions : include
    UC_ViewRegistrations ..> UC_ExportData : extend
